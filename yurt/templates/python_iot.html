NodeMCU Wi-Fi модуль үшін Arduino коды


#include <ESP8266WiFi.h> // Wi-Fi модулін қосу
#include <ESP8266HTTPClient.h>  // HTTP сұраныстарын жасау үшін кітапхана
#include <WiFiClientSecure.h>  // HTTPS арқылы қауіпсіз қосылу үшін
// Wi-Fi желісінің аты мен құпия сөзі
const char* ssid = "FiveHotel";
const char* password = "123456789";

// Мәліметтер жіберілетін сервердің адресі (API)
const char* serverUrl = "https://smartyurt.pythonanywhere.com/api/sensor/";

// SSL сертификатын тексермей қосылу үшін арнайы клиент
WiFiClientSecure client;

void setup() {
  Serial.begin(115200); // Сериялық мониторды іске қосу (дебаг үшін)
  WiFi.begin(ssid, password); // Wi-Fi-ға қосылу басталады

  // Wi-Fi қосылғанша күту
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Wi-Fi-ға қосылуда...");
  }

  Serial.println("Wi-Fi-ға сәтті қосылды!");

  // Сертификатты тексеруді өшіру (егер HTTPS болса, кейде қажет)
  client.setInsecure();
}

void loop() {
  // Егер Wi-Fi қосылған болса
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http; // HTTP клиентін жасау

    // Серверге қосылу (HTTPS клиент арқылы)
    http.begin(client, serverUrl);

    // Жіберілетін мәлімет форматы — JSON
    http.addHeader("Content-Type", "application/json");

    // Температура мен ылғалдылық бойынша мәліметтер (уақытша тесттік мәндер)
    String jsonData = "{\"temperature\": 25, \"humidity\": 60}";

    // POST сұранысы жіберіледі
    int httpResponseCode = http.POST(jsonData);

    // Серверден жауап алу
    if (httpResponseCode > 0) {
      Serial.print("Сервер жауабы: ");
      Serial.println(httpResponseCode);
      String response = http.getString();
      Serial.println("Мәлімет: " + response);
    } else {
      Serial.print("Сұраныс жіберу қатесі: ");
      Serial.println(httpResponseCode);
    }

    http.end(); // HTTP клиентін жабу
  }

  delay(60000); // 1 минут сайын мәліметтерді жіберу
}

Mega контроллер үшін Arduino коды


#include 

#define DHTPIN 7        // Датчикке қосылған пин (7-ші пин)
#define DHTTYPE DHT11   // DHT11 немесе DHT22 датчигін пайдалансаң, сол типті көрсету

DHT dht(DHTPIN, DHTTYPE); // DHT датчигін инициализациялау

void setup() {
  Serial.begin(9600); // Serial байланысы үшін бастау
  dht.begin();        // Датчикті іске қосу
}

void loop() {
  float temperature = dht.readTemperature(); // Температураны оқу
  float humidity = dht.readHumidity();       // Ылғалдылықты оқу

  if (isnan(temperature) || isnan(humidity)) {  // Егер оқу қатесі болса
    Serial.println("DHT датчигінен оқу қатесі!");
  } else {
    // Формат: temp=25.00;hum=60.00
    Serial.print("temp="); // Температураны шығару
    Serial.print(temperature); // Температура мәнін көрсету
    Serial.print(";hum=");     // Ылғалдылыққа дейін үтір қосу
    Serial.println(humidity);  // Ылғалдылықты шығару
  }

  delay(5000); // Әр 5 секунд сайын оқу
}

from flask import Flask, request, jsonify

app = Flask(__name__)

@app.route('/api/sensor', methods=['GET'])
def get_sensor_data():
    # Серверден алынған сенсор мәліметтерін көрсету
    data = {
        "temperature": 25,  # Бұл жерге нақты мәліметтерді қойыңыз
        "humidity": 60
    }
    return jsonify(data)

@app.route('/api/sensor/', methods=['POST'])
def receive_sensor_data():
    data = request.get_json()  # Деректерді алу
    temperature = data['temperature']
    humidity = data['humidity']

import json
from .models import SensorData

    return render(request, 'home.html')

    return render(request, 'business_plan.html')

def modeldeu(request):
    return render(request, 'modeldeu.html')

def koldanu(request):
    return render(request, 'koldanu.html')

def arduino(request):
    return render(request, 'arduino.html')

def iot(request):
    return render(request, 'iot.html')

def gallery(request):
    return render(request, 'gallery.html')

def iot_view(request):
    return render(request, 'iot.html')

def python_iot_view(request):
    return render(request, 'python_iot.html')

def weather_dashboard(request):
    data = SensorData.objects.order_by('-timestamp')[:10]
    return render(request, 'weather_dashboard.html', {'data': data})

@csrf_exempt
def receive_sensor_data(request):
    if request.method == 'POST':
        try:
            data = json.loads(request.body)
            temperature = data.get('temperature')
            humidity = data.get('humidity')

            if temperature is not None and humidity is not None:
                SensorData.objects.create(temperature=temperature, humidity=humidity)
                return JsonResponse({'status': 'success'})
            else:
                return JsonResponse({'status': 'error', 'message': 'Missing fields'}, status=400)
        except:
            return JsonResponse({'status': 'error', 'message': 'Invalid JSON'}, status=400)
    else:
        return JsonResponse({'status': 'error', 'message': 'Invalid method'}, status=405)

from django.db import models

class Feature(models.Model):
    title = models.CharField(max_length=200)
    description = models.TextField()
    image = models.ImageField(upload_to='features/')

    def __str__(self):
        return self.title

class SensorData(models.Model):

    humidity = models.FloatField()
