1. Arduino UNO арқылы RFID-оқығышты қалай іске қосады?

Arduino код:

#include      // Серво моторды басқару үшін кітапхана
#include     // HT1621 дисплейін басқару үшін кітапхана
#include        // SPI байланысын қолдану үшін кітапхана
#include    // RFID оқу модулін басқару үшін кітапхана

#define SS_PIN 10  // RFID модулінің SS (Slave Select) пині
#define RST_PIN 9  // RFID модулінің Reset (Қайта іске қосу) пині
#define CS 7       // HT1621 дисплейінің Chip Select (CS) пині
#define WR 6       // HT1621 дисплейінің Write (Жазу) пині
#define DATA 5     // HT1621 дисплейінің Data (Деректер) пині

String UID = "6CCB4322";  // RFID картасының дұрыс UID коды
byte lock = 0;  // Құлыптың күйі (0 - ашық, 1 - жабық)

Servo servo;  // Серво мотор объектісін анықтау
HT1621 lcd;   // HT1621 дисплей объектісін анықтау
MFRC522 rfid(SS_PIN, RST_PIN);  // RFID оқу модулін анықтау

void setup() {
  Serial.begin(9600);  // Сериялық байланыс жылдамдығын орнату
  servo.attach(3);     // Серво моторды 3-пинге қосу
  servo.write(70);     // Бастапқы күйін орнату (жабық есік)

  lcd.begin(CS, WR, DATA); // Дисплейді инициализациялау
  lcd.clear();             // Дисплейді тазалау
  lcd.print("HELLO");      // "HELLO" мәтінін көрсету (тестілеу үшін)

  SPI.begin();      // SPI байланысын бастау
  rfid.PCD_Init();  // RFID оқу модулін инициализациялау
}

void loop() {
  lcd.clear();
  lcd.print("CARD?");  // "CARD?" деген хабарламаны шығару
  delay(1000);

  if (!rfid.PICC_IsNewCardPresent()) return;  // Егер жаңа карта болмаса, қайта оралу
  if (!rfid.PICC_ReadCardSerial()) return;    // Егер картаны оқу мүмкін болмаса, қайта оралу

  lcd.clear();
  lcd.print("SCAN...");  // "SCAN..." хабарламасын шығару
  Serial.print("NUID tag is: ");
  String ID = "";

  // Карта UID кодын оқу
  for (byte i = 0; i < rfid.uid.size; i++) {
    ID.concat(String(rfid.uid.uidByte[i] < 0x10 ? "0" : ""));
    ID.concat(String(rfid.uid.uidByte[i], HEX));
    delay(300);
  }
  ID.toUpperCase();  // UID кодын бас әріптерге айналдыру

  // Егер дұрыс карта оқылса және есік жабық болса, оны ашу
  if (ID == UID && lock == 0) {
    servo.write(70);  // Есікті жабу
    lcd.clear();
    lcd.print("LOCKED");  // "LOCKED" хабарламасын шығару
    delay(1500);
    lock = 1;
  }
  // Егер дұрыс карта оқылса және есік жабық болмаса, оны жабу
  else if (ID == UID && lock == 1) {
    servo.write(160); // Есікті ашу
    lcd.clear();
    lcd.print("OPEN");  // "OPEN" хабарламасын шығару
    delay(1500);
    lock = 0;
  }
  // Егер UID сәйкес келмесе, қате хабарламасын шығару
  else {
    lcd.clear();
    lcd.print("ERROR");  // "ERROR" хабарламасын шығару
    delay(1500);
  }
}
2. TFD lCD mega shield арқылы температура мен ылғалдылықты экранға шығару

Arduino код:


#include <MCUFRIEND_kbv.h>   // TFT дисплейін басқару үшін кітапхана
#include <Adafruit_GFX.h>    // Графикалық элементтерді салу үшін кітапхана
#include <DHT.h>             // DHT11 температура және ылғалдылық сенсорын басқару кітапханасы
#include <Servo.h>           // Серво моторды басқару үшін кітапхана

#define DHTPIN 46       // DHT11 сенсоры қосылған пин
#define DHTTYPE DHT11   // Сенсордың түрі
#define SERVO_PIN 6     // Серво мотордың пині

DHT dht(DHTPIN, DHTTYPE);  // DHT сенсорын инициализациялау
MCUFRIEND_kbv tft;         // TFT дисплей объектісі
Servo myServo;             // Серво мотор объектісі

bool servoMoving = false;  // Серво мотордың қозғалыс флагы

void setup() {
    Serial.begin(115200); // Сериялық байланыс жылдамдығын орнату
    dht.begin();          // DHT сенсорын іске қосу
    myServo.attach(SERVO_PIN); // Серво моторды белгілі бір пинге қосу
    myServo.write(0);     // Серво моторды бастапқы күйге қою

    uint16_t ID = tft.readID(); // TFT дисплейдің идентификаторын оқу
    tft.begin(ID);              // TFT дисплейді іске қосу
    tft.setRotation(1);         // Экранның айналу бағытын орнату

    // Экранды көк түспен бояу
    tft.fillScreen(0x001F);
    tft.setTextColor(0xF800); // Қызыл түсті мәтін
    tft.setTextSize(3);       // Мәтіннің өлшемі
    tft.setCursor(50, 20);    // Мәтіннің координаталары
    tft.print("SMART YURT");  // Экранға "SMART YURT" жазуын шығару

    drawYurt(); // Юртаның суретін салу
}

void loop() {
    float h = dht.readHumidity();   // Ауаның ылғалдылығын оқу
    float t = dht.readTemperature(); // Температураны оқу

    if (isnan(h) || isnan(t)) { // Егер мәліметтер қате болса, циклдан шығу
        return;
    }

    // Экранда температура мен ылғалдылық мәндерін жаңарту
    tft.fillRect(20, 60, 200, 100, 0x001F); // Көне деректерді тазалау
    tft.setTextColor(0xFFFF); // Ақ түсті мәтін
    tft.setTextSize(2);
    tft.setCursor(40, 80);
    tft.print("Temp: "); tft.print(t); tft.print(" C"); // Температураны шығару
    tft.setCursor(40, 110);
    tft.print("Humidity: "); tft.print(h); tft.print(" %"); // Ылғалдылықты шығару

    // Егер температура 25°C-тан жоғары болса, сервопривод қозғалады
    if (t > 25) {
        if (!servoMoving) {
            servoMoving = true;
            moveServo(); // Серво моторды қозғалысқа келтіру
        }
    } else {
        myServo.write(0); // Егер температура төмендесе, серво моторды бастапқы күйге қою
        servoMoving = false;
    }

    delay(2000); // 2 секунд күту
}

// Серво моторды қозғалысқа келтіретін функция
void moveServo() {
    for (int pos = 0; pos <= 180; pos += 10) { // 0-ден 180 градусқа дейін бұру
        myServo.write(pos);
        delay(100);
    }
    for (int pos = 180; pos >= 0; pos -= 10) { // 180-ден 0 градусқа дейін кері бұру
        myServo.write(pos);
        delay(100);
    }
}

// TFT дисплейде юрта суретін салу функциясы
void drawYurt() {
    tft.fillCircle(160, 180, 40, 0x964B00); // Юртаның күмбезін салу
    tft.fillRect(120, 180, 80, 50, 0x964B00); // Юртаның қабырғасын салу
    tft.drawLine(120, 180, 160, 140, 0xFFFF); // Юртаның оң жақ бағанын салу
    tft.drawLine(200, 180, 160, 140, 0xFFFF); // Юртаның сол жақ бағанын салу
    tft.fillRect(150, 200, 20, 30, 0x0000); // Юртаның есігін қара түспен салу
}
